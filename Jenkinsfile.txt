pipeline {
    agent any

    tools {
        // Make sure Maven is configured in Jenkins (Global Tool Configuration)
        maven 'Maven'
    }

    stages {

        stage('Checkout Code') {
            steps {
                // If using GitHub
                git 'https://github.com/Vijay-Kumar2309/simple-app-java.git'
            }
        }

        stage('Build & Run Tests') {
            steps {
                // For Windows (since you're using 'bat')
                bat 'mvn clean test'
            }
        }

        stage('Package') {
            steps {
                bat 'mvn package'
            }
        }

    }

    post {
        success {
            echo 'Build SUCCESS'
        }
        failure {
            echo 'Build FAILED'
        }
    }
}
